  <script>
    // Notice scrolling script
    window.addEventListener('DOMContentLoaded', () => {
      const noticeList = document.getElementById('noticeList');
      if (!noticeList) return; // Safety check

      const itemCount = noticeList.children.length;
      if (itemCount === 0) return;

      const itemHeight = noticeList.children[0].offsetHeight;
      const totalHeight = itemCount * itemHeight;
      const duration = itemCount * 3;

      const styleSheet = document.createElement('style');
      styleSheet.innerHTML = `
        @keyframes roll-up {
          0% { top: 0; }
          100% { top: -${totalHeight}px; }
        }
      `;
      document.head.appendChild(styleSheet);

      noticeList.style.position = "relative";
      noticeList.style.top = "0";
      noticeList.style.animation = `roll-up ${duration}s linear infinite`;
    });
  </script>
